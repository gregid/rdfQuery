 65     getValueFromElement = function () {
 66         var tag = this.get(0).nodeName.toLowerCase();
 67         var value;
 68         if (tag === 'meta') {
 69             return $.rdf.literal(this.attr('content') === undefined ? '""' : this.attr('content'));
 70         } else if (tag === 'audio' || 
 71             tag === 'embed' || 
 72             tag === 'iframe' || 
 73             tag === 'img' || 
 74             tag === 'source' || 
 75             tag === 'track' || 
 76             tag === 'video') {
 77                 if (this.attr('src')) {
 78                     return $.rdf.resource(resolveURL(this.attr('src')));
 79                 } else {
 80                    return $.rdf.resource('<>');
 81                 }
 82         } else if (tag === 'a' || 
 83             tag === 'area' || 
 84             tag === 'link') {
 85                 if (this.attr('href')) {
 86                     return $.rdf.resource(resolveURL(this.attr('href')));
 87                 } else {
 88                    return $.rdf.resource('<>');
 89                 }
 90         } else if (tag === 'object') {
 91                 if (this.attr('data')) {
 92                     return $.rdf.resource(resolveURL(this.attr('data')));
 93                 } else {
 94                    return $.rdf.resource('<>');
 95                 }
 96         } else if (tag === 'time' && this.attr('datetime') !== undefined) {
 97            if (this.attr('datetime')) {
 98                 return $.rdf.literal('"' + this.attr('datetime') + '"');
 99             } else {
100                return $.rdf.literal('""');
101             }
102         } else {
103             return $.rdf.literal('"' + this.text() + '"');
104         }
105     },

107     extractProperties = function (context) {
108         var triples = [];
109                 
110         //there are two types of properties
111         // (1 - the easy ones): literals
112         var property = getPropertyFromElement.call(this);
113         if (property !== undefined) {
114             if (this.attr('itemscope') === undefined) {
115                 var value = getValueFromElement.call(this);
116                 
117                 var triple = $.rdf.triple(
118                     context.subject,
119                     property,
120                     value, {namespaces: ns});
121                 
122                 triples.push(triple);
123             }
124             else {
125                 // (2 - the more complex ones): resources
126                 //recursively call extractResource to parse the triples
127                                 
128                 var res = extractResource.call(this, {
129                     subject: context.subject,
130                     property: property
131                 });
132                 triples = triples.concat(res.triples);
133                 return triples;
134             }
135         }
136         
137         //search for children with properties
138         var propertyTriples = $.map(this.children(), function (elem) {
139             return extractProperties.call($(elem), context);
140         });
141         triples = triples.concat(propertyTriples);
142         return triples;
143     },

145     extractResource = function (context) {
146         var triples = [];
147         
148         if (!context) {
149             context =  {};
150         }
151         
152         var subject = getSubjectFromElement.call(this);
153         //add possible back-reference
154         if (context.subject && context.property) {
155             triples.push(
156                 $.rdf.triple(
157                     context.subject, 
158                     context.property, 
159                     subject, {namespaces: ns}));
160         }
161        
162        
163         //get type of element
164         var type = getTypeFromElement.call(this);
165         triples.push(
166                 $.rdf.triple(
167                     subject, 
168                     $.rdf.type, 
169                     type, {namespaces: ns}));
170        
171        //query for referenced items and add triples recursively
172         if (this.attr('itemref') !== undefined) {
173             var selector = $.map(this.attr('itemref').split(" "), 
174             function (n, i) {
175                 return "#" + n;
176             })
177             .join(", ");
178             var referencedProperties = extractProperties.call($(selector), {
179                 subject: subject
180             });
181             triples = triples.concat(referencedProperties);
182         }
183                
184         //parse children for properties (recursively)!
185         var propertyTriples = $.map(this.children(), function (elem) {
186             return extractProperties.call($(elem), {subject: subject});
187         });
188         triples = triples.concat(propertyTriples);
189         
190         return {
191             triples: triples
192         };
193     },

209       if (options && options.about !== undefined) {
210           if (options.about === null) {
211               return this.attr('itemprop') !== undefined ||
212                      this.attr('itemref') !== undefined ||
213                      this.attr('itemtype') !== undefined;
214           } else {
215               return getSubjectFromElement.call(this) === options.about;
216           }
217       } else if (options && options.type !== undefined) {
218           var type = getTypeFromElement.call(this);
219           if (type !== undefined) {
220               return options.type === null ? true : (type === options.type);
221           }
222           return false;
223       } else {
224           return microdata.call(this, options);
225       }

238   /**
239    * Creates a {@link jQuery.rdf} object containing the RDF triples parsed from the microdata found in the current 
240    * jQuery selection or adds the specified triple as microdata markup on each member of the current jQuery selection.
241    * To create an {@link jQuery.rdf} object, you will usually want to use {@link jQuery#rdf} instead, as this may 
242    * perform other useful processing (such as of microformats used within the page).
243    * @methodOf jQuery#
244    * @name jQuery#microdata
245    * @param {jQuery.rdf.triple} [triple] The RDF triple to be added to each item in the jQuery selection.
246    * @returns {jQuery.rdf}
247    * @example
248    * // Extract microdata markup from all span elements contained inside #main
249    * rdf = $('#main > span').microdata();
250    * @example
251    * // Add microdata markup to a particular element
252    *  var span = $('#main > p > span');
253    *  span.microdata('<> name "Test Name" .');
254    */
255   $.fn.microdata = function (triple, ns) {
256     if (triple === undefined) {
257       var triples = $.map($(this), function (elem) {
258         return microdata.call($(elem));
259       });
260       return $.rdf({ triples: triples });
261     } else {
262       $(this).each(function () {
263         addMicrodata.call($(this), triple, ns);
264       });
265       return this;
266     }
267   };

271    * can be either an object or an array of objects. The objects can either have a <code>type</code> property, in which 
272    * case the specified type is removed from the microdata provided on the selected elements, or a <code>property</code> property,
273    * in which case the specified property is removed from the microdata provided on the selected elements.
274    * @methodOf jQuery#
275    * @name jQuery#removeMicrodata
276    * @param {Object|Object[]} triple The microdata markup items to be removed
277    * from the items in the jQuery selection.
278    * @returns {jQuery} The original jQuery object.
279    * @example 
280    * // To remove a property resource or relation from an element 
281    * $('#main > p > a').removeMicrodata({ itemprop: "creator" });
282    * @example
283    * // To remove a type from an element
284    * $('#main >p > a').removeMicrodata({ itemtype: "http://schema.org/Person" });
285    * @example
286    * // To remove multiple triples from an element
287    * $('#main > p > a').removeMicrodata([{ itemprop: "depicts" }, { itemprop: "creator" }]);
288    */
289   $.fn.removeMicrodata = function (triple) {
